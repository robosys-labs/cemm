**Goal:** Make a small, focused step toward full NER by adding a rule-based named entity extractor in `SemanticInterpreter`. It should recognize capitalized proper noun phrases, temporal expressions, and numbers as entity refs when the UOL mapper does not already provide them.

**Architecture:** `SemanticInterpreter._build_graph` already infers entity refs from causal/temporal processes when none are present. This task adds a complementary `_extract_named_entities` pass that scans the raw text for named entities and adds them as `entity_ref` atoms.

In `SemanticInterpreter`, add:

```python
def _extract_named_entities(self, content: str) -> list[dict[str, Any]]:
    """Extract simple named entities from content when UOL mapper provides none."""
    entities: list[dict[str, Any]] = []
    seen: set[str] = set()
    words = content.split()
    stop_words = {"The", "A", "An", "And", "Or", "But"}

    # Capitalized proper-noun phrases (naive)
    i = 0
    while i < len(words):
        if words[i][0].isupper() and words[i] not in stop_words:
            phrase = [words[i]]
            j = i + 1
            while j < len(words) and words[j][0].isupper() and words[j] not in stop_words:
                phrase.append(words[j])
                j += 1
            entity_text = " ".join(phrase).lower()
            if entity_text not in seen:
                entities.append({
                    "kind": "entity_ref",
                    "entity_id": entity_text,
                    "role": "entity",
                    "confidence": 0.6,
                })
                seen.add(entity_text)
            i = j
        else:
            i += 1

    # Temporal expressions
    temporal_words = {"today", "tomorrow", "yesterday", "now", "later", "soon", "morning", "afternoon", "evening", "night"}
    for w in words:
        wl = w.lower().strip(".,!?;")
        if wl in temporal_words and wl not in seen:
            entities.append({
                "kind": "entity_ref",
                "entity_id": wl,
                "role": "time",
                "confidence": 0.5,
            })
            seen.add(wl)

    return entities
```

```python
def test_named_entities_extracted_for_proper_nouns():
    from cemm.registry.registry import Registry
    from cemm.registry.uol_mapper import UOLMapper
    from cemm.kernel.semantic_interpreter import SemanticInterpreter
    from cemm.types.signal import Signal, SignalKind, SourceType
    from cemm.types.context_kernel import ContextKernel
    from cemm.types.permission import Permission

    registry = Registry()
    mapper = UOLMapper(registry)
    interpreter = SemanticInterpreter(mapper, store=None)
    kernel = ContextKernel(id="ctx", permission=Permission.public())
    signal = Signal(id="s1", kind=SignalKind.TEXT, source_id="u", source_type=SourceType.USER, content="Alice visited Paris last summer", observed_at=0.0, context_id="ctx")
    seg = interpreter.run(signal, kernel)
    entity_ids = {ref.get("entity_id", ref.get("entity", "")) for ref in seg.entity_refs}
    assert "alice" in entity_ids
    assert "paris" in entity_ids
```
